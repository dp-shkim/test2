git status    		: 현재 파일들 상태

git add	  		: 파일저장

git rm 			: add 취소

git commit -m "테스트" 	: 

git commit -am "테스트"	: add, commit 한번에

git log			: commit 내역

git remote		: 원격 저장소 확인

git remote add shkim 주소	: 원격 저장소 주소 등록

git remote remove shkim	: 원격 저장소 삭제

git push shkim master	: 원격 저장소에 commit 저장

git pull shkim master	: 다른 사람 업데이트 내역 PC로 받기

git clone 주소		: 서버 프로젝트 받기

git diff			: 수정된 파일 기존과 비교

git checkout test.txt	: 수정된 파일 되돌리기

git reset test.txt		: 파일 올리기 전으로 되돌리기
(되돌리기 옵션 몇 개 있음)

git revert			: commit 내용을 새 commit으로 저장

git branch 		: 현재 branch 확인

git branch 이름		: 새 branch 생성

git branch -d 이름		: branch 삭제

git checkout 브랜치 이름	: 새 branch로 HEAD 이동


git push --delete shkim 브랜치 이름 : 원격 저장소 브랜치 삭제
